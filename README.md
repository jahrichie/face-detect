# Requires python 3.11 (not .12)


## Run the following commands for Ubuntu
- python3 -m venv --system-site-packages face-detect
- source face-detect/bin/activate
- sudo apt update
- sudo apt full-upgrade
- From here you can either run the install-face-stuff.sh or the below commands
- pip install opencv-python
- pip install imutils
- sudo apt install cmake
- pip install face-recognition


## To Run On raspberry pi os
- Run the files that have pi_os in them instead of core files
